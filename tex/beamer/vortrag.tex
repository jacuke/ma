\section{Versionsübergreifende Umsteiger-Suche}

\begin{frame}
  \frametitle{Umsteiger-Suche -- Versionsübergreifend}
  Problematik:
    \begin{itemize}
        \item Die einzelnen Datensätze beziehen sich immer auf die neue Version und die Vorgängerversion
		\item Wie können Umsteiger über alle Versionen ermittelt werden?
		\item Ausgehend von einer beliebigen Version chronologisch vorwärts und rückwärts
    \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Umsteiger-Suche -- Horizontal}
  Vorgehensweise:
	\begin{itemize}
		\item Ausgehend von einem Kode einer Version, wird im Vergleich mit der jeweils zeitlich neueren und älteren Version gesucht, ob es Umsteiger gibt
        \item Solange bis die neueste und älteste Version erreicht ist
		\item Falls Umsteiger gefunden werden, wird die Suche mit diesen in die gleiche Richtung fortgesetzt, statt des ursprünglichen Kodes
		\item Jedes Umsteiger-Ergebnis wird in eine sich rekursiv aufbauende Datenstruktur aufgenommen
    \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Umsteiger-Suche -- Vertikal}
  \emph{Purdom's Algorithm}: Effizient für die Bestimmung der transitiven Hülle eines gerichteten Graphen. \newline \\
  
  Zwei Hauptmerkmale:
      \begin{itemize}
        \item Topologisch umgekehrte Vorgehensweise: Weil später ermittelte Nachbarn zu den vorher bearbeiteten Knoten einfach hinzugefügt werden können
        \item Vereinigung von stark zusammenhängenden Komponenten in je einen Knoten als Zwischenschritt (ergibt einen zyklenfreien Graphen)
    \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Umsteiger-Suche -- Vergleich der zwei Verfahren}
  Horizontal
      \begin{itemize}
        \item Schneller für einen Kode
		\item Einfacher zusätzliche Informationen auszulesen
        \item Anwendung: Anzeige der Suchergebnisse
    \end{itemize}
	Vertikal
	\begin{itemize}
        \item Wesentlich schneller für alle Kodes einer Version
        \item Anwendung: \begin{itemize}
        \item Generierung der ConceptMaps
        \item Bestimmung Umsteiger ja/nein
    \end{itemize}
    \end{itemize}
\end{frame}

  \item Aber unterschiedliche Bearbeitung jedes Nachrichtentyps
  \item Manche Parameter der FHIR Ressourcen ergeben sich lediglich aus dem Kontext der V2 Nachrichten
  \item Mapping der V2 Coding Tables auf FHIR Value Sets teilweise (noch) nicht vorhanden oder inkonsistent
  \item Einige (redundante) Informationen in den V2 Segmenten benötigen nur eine Referenz auf schon bestehende FHIR Ressourcen (z.B. internes Personal/Locations)
  \end{itemize}
}
